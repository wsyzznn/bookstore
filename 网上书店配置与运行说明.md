# 网上书店配置与运行说明

共分为三个步骤：

* 数据库启动
* 后端启动
* 前端启动

接下来分步说明。

## 数据库启动

1. 创建MySQL数据库bookstore

```sql
CREATE DATABASE bookstore
CHARACTER SET utf8mb4 
COLLATE utf8mb4_unicode_ci;
```

2. 创建实体表和视图

```sql
CREATE TABLE `users` ( 
`id` BIGINT AUTO_INCREMENT PRIMARY KEY, 
`username` VARCHAR(50) NOT NULL UNIQUE, 
`password` VARCHAR(100) NOT NULL, 
`email` VARCHAR(100), 
`balance` DECIMAL(10, 2) DEFAULT 99999.00, 
  `role` VARCHAR(20) NOT NULL DEFAULT 'CUSTOMER', -- 'CUSTOMER', 'MERCHANT', 'ADMIN' 
  `create_time` TIMESTAMP DEFAULT CURRENT_TIMESTAMP 
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `books` ( 
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY, 
  `seller_id` BIGINT NOT NULL, -- 关联到发布该书的商家 ID 
  `isbn` VARCHAR(20) NOT NULL UNIQUE, 
  `title` VARCHAR(100) NOT NULL, 
  `author` VARCHAR(100) NOT NULL, 
  `category` VARCHAR(50) DEFAULT 'General', -- 新增分类字段 
  `price` DECIMAL(10, 2) NOT NULL, 
  `stock_quantity` INT NOT NULL DEFAULT 100, 
  `cover_image` VARCHAR(255), 
  `description` TEXT, 
  FOREIGN KEY (`seller_id`) REFERENCES `users` (`id`) 
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `book_ratings` ( 
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY, 
  `user_id` BIGINT NOT NULL, 
  `book_id` BIGINT NOT NULL, 
  `rating` TINYINT NOT NULL COMMENT '1-5星', 
  `create_time` TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
  UNIQUE KEY `uk_user_book` (`user_id`, `book_id`), 
  FOREIGN KEY (`user_id`) REFERENCES `users` (`id`), 
  FOREIGN KEY (`book_id`) REFERENCES `books` (`id`) 
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `orders` ( 
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY, 
  `user_id` BIGINT NOT NULL, 
  `total_price` DECIMAL(10, 2) NOT NULL, 
  `recipient_name` VARCHAR(50) NOT NULL, -- 收件人快照 
  `recipient_phone` VARCHAR(20) NOT NULL, -- 电话快照 
`shipping_address` VARCHAR(255) NOT NULL, -- 地址快照 
`status` VARCHAR(20) DEFAULT 'SUCCESS', -- 已移除发货流程，默认为 SUCCESS 
`create_time` TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
FOREIGN KEY (`user_id`) REFERENCES `users` (`id`) 
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4; 

CREATE TABLE `order_items` ( 
`id` BIGINT AUTO_INCREMENT PRIMARY KEY, 
`order_id` BIGINT NOT NULL, 
`book_id` BIGINT NOT NULL, 
`quantity` INT NOT NULL, 
`price_at_purchase` DECIMAL(10, 2) NOT NULL, 
FOREIGN KEY (`order_id`) REFERENCES `orders` (`id`), 
FOREIGN KEY (`book_id`) REFERENCES `books` (`id`) 
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `addresses` ( 
`id` BIGINT AUTO_INCREMENT PRIMARY KEY, 
`user_id` BIGINT NOT NULL, 
`recipient_name` VARCHAR(50) NOT NULL COMMENT '收件人姓名', 
`phone_number` VARCHAR(20) NOT NULL COMMENT '联系电话', 
`address_detail` VARCHAR(255) NOT NULL COMMENT '详细地址', 
`create_time` TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
FOREIGN KEY (`user_id`) REFERENCES `users` (`id`) 
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4; 

CREATE VIEW v_books_public AS 
SELECT b.id AS bookId, b.title, b.author, b.isbn, b.category, b.price, b.stock_quantity AS 
stockQuantity, b.cover_image AS coverImage, b.description, b.seller_id AS sellerId, 
(SELECT IFNULL(AVG(rating), 0) FROM book_ratings br WHERE br.book_id = b.id) AS 
averageRating 
FROM books b 
WHERE b.stock_quantity > 0; 

CREATE VIEW v_order_detail AS 
SELECT  
    o.id AS orderId,  
    o.user_id AS userId,  
    o.create_time AS createTime,  
    o.total_price AS totalPrice, 
    o.status, 
    o.recipient_name AS recipientName,  
    o.recipient_phone AS recipientPhone, 
    o.shipping_address AS shippingAddress, 
    oi.book_id AS bookId,  
    b.title AS bookTitle,  
    oi.quantity,  
    oi.price_at_purchase AS priceAtPurchase, 
    (oi.quantity * oi.price_at_purchase) AS itemSubTotal,
    b.cover_image AS bookCover, 
    b.seller_id AS sellerId
FROM orders o 
JOIN order_items oi ON o.id = oi.order_id 
JOIN books b ON oi.book_id = b.id;

CREATE OR REPLACE VIEW v_order_detail_for_entity AS
SELECT
    o.id AS order_id,
    o.user_id AS user_id,
    o.create_time AS create_time,
    o.total_price AS total_price,
    o.status,
    o.recipient_name AS recipient_name,
    o.recipient_phone AS recipient_phone,
    o.shipping_address AS shipping_address,
    oi.book_id AS book_id,
    b.title AS book_title,
    oi.quantity,
    oi.price_at_purchase AS price_at_purchase,
    (oi.quantity * oi.price_at_purchase) AS item_sub_total,
    b.cover_image AS book_cover,
    b.seller_id AS seller_id
FROM orders o
JOIN order_items oi ON o.id = oi.order_id
JOIN books b ON oi.book_id = b.id;

CREATE VIEW v_user_orders AS 
SELECT o.id AS orderId, o.user_id AS userId, o.create_time AS createTime, o.total_price AS 
totalPrice, o.status, 
COUNT(oi.book_id) AS bookCount 
FROM orders o 
JOIN order_items oi ON o.id = oi.order_id 
GROUP BY o.id, o.user_id, o.create_time, o.total_price, o.status; 

CREATE VIEW v_inventory_status AS 
SELECT id AS bookId, title, stock_quantity AS stockQuantity, 
CASE 
WHEN stock_quantity = 0 THEN 'Out of Stock' 
WHEN stock_quantity < 5 THEN 'Low Stock' 
ELSE 'In Stock' 
END AS stockStatus, 
seller_id AS sellerId 
FROM books; 

CREATE VIEW v_users_safe AS 
SELECT id, username, role, email FROM users;
```

## 后端启动

```
cd bookstore\bookstorebackend
mvn spring-boot:run
```

## 前端启动

需先下载Node.js（建议 18+）

```
cd bookstore\online-bookstore-dev\online-bookstore-frontend
npm install
npm run dev
```

成功启动后，访问http://localhost:5173/
